        // 두 숫자끼리 비교하기 위해 배열 길이 -1 만큼 설정
        for (let i = 0; i < arr.length -1; i++) {
            // j를 i보다 1 크게 만들어 두 숫자를 비교
            for (let j = 1; j < arr.length; j++) {
                // 비교하는 앞 숫자가 더 크면 왼쪽으로 한 칸 이동
                if (arr[i] > arr[j]) {
                    let temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                }
            }
            console.log(arr)
        }
        return arr;
    }
